#include <iostream>

using namespace std;

int main()
{
    double a=10.5, b=20.3;
    short c = 30;
    cout << " 1. double zajmuje "<< sizeof(double) << "B" << endl;
    cout << " 2. short  zajmuje "<< sizeof(short) << "B" << endl;
    cout << " 3. Adres a: "<< &a << endl;
    cout << " 4. Adres b: "<< &b << endl;
    cout << " 5. Adres c: "<< &c << endl;
    cout << endl;

    double *p1 = NULL, *p2 = nullptr; // zalecane jest nullptr
    short *p3 = &c;
    cout << " 6. double* zajmuje "<< sizeof(double*) << "B" << endl;
    cout << " 7. short*  zajmuje "<< sizeof(short*) << "B" << endl;
    p1 = &a;
    p2 = &b;
    cout << " 8. p1: "<< p1 << "\t *p1: " << *p1 << endl;
    cout << " 9. p2: "<< p2 << "\t *p2: " << *p2 << endl;
    cout << "10. p3: "<< p3 << "\t *p3: " << *p3 << endl;
    p2 = p1;
    cout << "11. p2: "<< p2 << "\t *p2: " << *p2 << endl;
    *p1 = 50;
    cout << "12. p1: "<< p1 << "\t *p1: " << *p1 << endl;
    cout << "13. p2: "<< p2 << "\t *p2: " << *p2 << endl;
    cout << "14. &a: "<< &a << "\t   a: " << a << endl;

    void *p4 = nullptr; //wskaznik bez typu
    p4 = (void *)&c;
    cout << "15. p4: "<< p4 << "\t *p4: " << *(short *)p4 << endl;

    return 0;
}
