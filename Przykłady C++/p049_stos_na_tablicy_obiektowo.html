#include <iostream>
#include <string>

using namespace std;

class MyStack
{
public:
    unsigned stackSize;
    int *items; // tablica elementow stosu
    int top; // indeks wierzcholka stosu

    MyStack(unsigned newSize):stackSize(newSize),top(newSize)
    {
        items = new int[stackSize];
    }

    ~MyStack()
    {
        delete items;
    }

    void push(int item);  //wartosc item na stos
    int pop(); // zdejmij ze stosu i zwroc
    void print(); // wypisz stos od top do dna
    void clear() { top = stackSize;}
    bool isEmpty() { return top == stackSize; }
};

void MyStack::push(int item)
{
    if (top > 0) {
        --top;
        items[top] = item;
    } else {
        cout << "Stos jest pelny! Operacja nie moze byc wykonana!" << endl;
    }
}

int MyStack::pop()
{
   if (top < stackSize){
        return items[top++];
   } else {
        cout << "Stos jest pusty! Operacja nie moze byc wykonana!" << endl;
    }

}



void MyStack::print()
{
    cout << endl;
    for(unsigned i = top; i < stackSize; ++i){
        cout << "\t| " << items[i] << "\t|"<< endl;
    }
    cout << "\t---------" << endl;
    cout << endl;

}


int main()
{

    MyStack s(20);
    for(int i = 0; i < 5; i++){
        s.push((i+1) * 10);
    }

    s.print();

    s.pop();
    s.print();

    s.clear();

    int dec = 123;

    // dopisz kod zamiana dec na bin z wykorzystaniem stosu s

    string bin = "";
    //     to_string( liczba_int )  -zwaraca string z int


    cout << dec << " = " << bin << endl;
    return 0;
}
