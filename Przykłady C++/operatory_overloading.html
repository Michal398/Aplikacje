#include <iostream>

using namespace std;

class Point
{
protected:
    int x;
    int y;
public:
    Point() : x(0), y(0) {}
    Point(int x, int y) : x(x), y(y) {}
    int getx() const { return x; }
    int gety() const { return y; }

    Point operator+(const Point p)
    {
        Point result;
        result.x = this->x + p.x;
        result.y = this->y + p.y;
        return result;
    }

    void operator+=(const Point p)
    {
        this->x = this->x + p.x;
        this->y = this->y + p.y;
    }

    friend std::ostream& operator<<(std::ostream&, const Point);
};

std::ostream& operator<<(std::ostream&, const Point);

int main()
{
    Point A(5, 6);
    Point B(2, 3);
    Point C = A + B;
    cout << " 1. " << A << endl;
    cout << " 2. " << B << endl;
    cout << " 3. " << C << endl;
    A += C;
    cout << " 4. " << A << endl;
    return 0;
}

std::ostream& operator<<(std::ostream& outStream, const Point p)
{
    outStream << "(" << p.x << "," << p.y << ")";
    //outStream << "(" << p.getx() << "," << p.gety() << ")";

    return outStream;
}
