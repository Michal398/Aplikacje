#include <iostream>
using namespace std;

void zadanie0(int n, char wersja);

int main()
{
    zadanie0(5,'w');
    zadanie0(5,'d');
    zadanie0(5,'f');

    zadanie0(-1,'f');
    zadanie0(5,'x');


    cout << "KONIEC" << endl;
    return 0;
}

void zadanie0(int n, char wersja)
{
    if( n <= 0 )
    {
        cout << "Nieprawidlowy pierwszy parametr" << endl;
        cout << endl;
        return;
    }
    if(wersja == 'w')
    {
        cout << "Zadanie nr 0 (while)" << endl;
        int i = 0;
        int j;
        while( i < n )
        {
            j = 0;
            while( j < n )
            {
                cout << "X ";
                ++j;
            }
            cout << endl;
            ++i;
        }
    }
    else if(wersja == 'd')
    {
        cout << "Zadanie nr 0 (do)" << endl;
        int i = 0;
        int j;
        do
        {
            j = 0;
            do
            {
                cout << "X ";
                ++j;
            }
            while( j < n );
            cout << endl;
            ++i;
        }
        while( i < n );
    }
    else if(wersja == 'f')
    {
        cout << "Zadanie nr 0 (for)" << endl;
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < n; j++)
            {
                cout << "X ";
            }
            cout << endl;
        }
    }
    else
    {
        cout << "Nieprawidlowy drugi parametr" << endl;
        cout << endl;
    }

    cout << endl;
}
