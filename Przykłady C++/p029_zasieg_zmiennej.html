#include <iostream>
using namespace std;

int c = 30;
int n = 1;

void f(void);

int main()
{
    int a = 1;
    int n = 2;
    std::cout << " 1. a = " << a << std::endl;  // 1
    cout << " 2. n = " << n << endl;  // 2
    {
        cout << " 3. n = " << n << endl;  // 2
        int b = 10;
        int n = 3;
        cout << " 4. a = " << a << endl; // 1
        cout << " 5. b = " << b << endl; // 10
        cout << " 6. c = " << c << endl; // 30
        cout << " 7. n = " << n << endl; // 3
        cout << " 8. n = " << ::n << endl; // 1
    }
    // cout << " 9. b = " << b << endl; // error
    cout << "10. c = " << c << endl; // 30
    cout << "11. n = " << n << endl; // 2
    f();
    return 0;
}

void f(void)
{
    cout << "f1. c = " << c << endl; // 30
    cout << "f2. n = " << n << endl; // 1
}
