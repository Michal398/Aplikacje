/** p012_string_cin_getline **/

#include <iostream>
#include <string>

using namespace std;

int main()
{
    string napis(""); // pusty string
    cout << "1. Wpisz tekst:";
    cin >> napis; //wczyta tekst do spacji
    cout << napis << endl;

    cin.clear(); //czyszczenie flag bledow
    cin.ignore( 1000, '\n' ); //usuniecie do 1000 znakow ze strumienia
                              //do znaku nowego wiersza '\n'

    cout << "2. Wpisz tekst:";
    getline(cin, napis); // wczytanie wiersza tekstu
    cout << napis << endl;

    return 0;
}
