/** p024_operatory_przypisania **/

#include <iostream>

using namespace std;

int main()
{
    int a, b;
    a = b = 5;
    cout << " 1. a: " << a << "  b: " << b << endl;
    a += 2;  // a = a + 2;
    b += a - 1;  // b = b + (a - 1);
    cout << " 2. a: " << a << "  b: " << b << endl;
    a -= 1;
    b *= 3 + 7;  // b = b * (3 + 7);
    cout << " 3. a: " << a << "  b: " << b << endl;
    a *= 3 + 1 + a;  // a = a * (3 + 1 + a);
    cout << " 4. a: " << a << endl;
    a %= 4;
    b /= 2;
    cout << " 5. a: " << a << "  b: " << b << endl;
    return 0;
}

<<=   >>=       Compound assignment by bitwise left shift and right shift
