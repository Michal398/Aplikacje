/** p007_operatory **/

#include <iostream>

using namespace std;

int main()
{
    cout << " 1. " << 8 % 3 << endl;
    // % - operator reszta z dzielenia (modulo)

/*
    Priorytety w wyrazeniach arytmetycznych
    P1  ( )  najwyzszy priorytet w wyrazeniu
    P2  + -     operator znaku liczby (operator unarny, czyli jednoargumentowy) )
    P3  * / %
    P4  + -  operatory binarne (dwuargumentowe)

*/
    cout << " 2. " << 2 * 3 % 5 << endl;
    int i,j,k,x;
    i = 100;
    j = 25;
    k = 13;
    x = (5 * ((j % k) + i) / (2 * k)) / 2;
    //    (5 * ((25 % 13) + 100) / (2 * 13)) / 2 =
    //  = (5 * (12 + 100) / 26) / 2 =
    //  = (5 * 112 / 26) / 2 =
    //  = (560 / 26) / 2 =
    //  = 21 / 2 = 10  <-- dzielenie calkowite

    cout << " 3. " << "x = " << x << endl;
    return 0;
}
