/** p011_string_c-string_escape_sequences **/

#include <iostream>
#include <string>

using namespace std;

int main()
{
    char cNapis1[] = "To jest c-string";
    char *cNapis2 = "To takze jest c-string";
    cout << cNapis1 << endl << cNapis2 << endl;

    string napis1 = "C++";
    string napis2("Jezyk");
    cout << napis2 << " " << napis1 << endl;
    string s = napis2 + " programowania " + napis1;
    cout << s << endl;
    //s = "Ala " + "ma kota."; // Zle!
    // "Ala" - null-terminated string, C-string
    // "Ala" --->  'A' 'l' 'a' '\0'
    // string("Ala")  - string
    s = "Ala " + string("ma kota.");
    cout << s << endl;

    //s = cNapis1 + " " + cNapis2; // Zle!
    s = string(cNapis1) + " " + cNapis2;
    cout << s << endl;

    // Escape sequences
    cout << "\n------------\n";
    s = "\aAla\nma kota\n\t\tFilemona";
    cout << s << endl;
    cout << "Ala ma kota \rEla\n";

    return 0;
}
